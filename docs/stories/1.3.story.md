**Trigger de validaĂ§ĂŁo de nĂ­vel:**

```sql
CREATE OR REPLACE FUNCTION check_level_limit()
RETURNS TRIGGER AS $$
DECLARE
  is_epic BOOLEAN;
  max_level INTEGER;
BEGIN
  SELECT epic_mode INTO is_epic FROM games WHERE id = NEW.game_id;
  max_level := CASE WHEN is_epic THEN 10 ELSE 20 END;

  IF NEW.level < 1 OR NEW.level > max_level THEN
    RAISE EXCEPTION 'Level % is out of valid range (1..%) for this game mode',
      NEW.level, max_level;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_check_level_limit
  BEFORE INSERT OR UPDATE ON game_players
  FOR EACH ROW EXECUTE FUNCTION check_level_limit();
```
